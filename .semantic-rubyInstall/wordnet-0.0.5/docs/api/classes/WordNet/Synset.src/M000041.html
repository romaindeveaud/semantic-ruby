                                def frames
                                        frarray = self.frameslist.split( WordNet::SUB_DELIM_RE )
                                        verbFrames = []

                                        @mutex.synchronize( Sync::SH ) {
                                                frarray.each {|fr|
                                                        fnum, wnum = fr.split
                                                        if wnum > 0
                                                                wordtext = " (" + self.words[wnum] + ")"
                                                                verbFrames.push VERB_SENTS[ fnum ] + wordtext
                                                        else
                                                                verbFrames.push VERB_SENTS[ fnum ]
                                                        end
                                                }
                                        }

                                        return verbFrames
                                end
