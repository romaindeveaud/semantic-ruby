                def inspect
                        pointer_counts = self.pointer_map.collect {|type,ptrs|
                                "#{type}s: #{ptrs.length}"
                                }.join( ", " )

                                %q{#<%s:0x%08x/%s %s (%s): "%s" (%s)>} % [
                                self.class.name,
                                self.object_id * 2,
                                self.offset,
                                self.words.join(", "),
                                self.part_of_speech,
                                self.gloss,
                                pointer_counts,
                        ]
                end
