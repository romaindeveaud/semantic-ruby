                                def |( otherSyn )

                                        # Find all of this syn's hypernyms
                                        hyperSyns = self.traverse( :hypernyms )
                                        commonSyn = nil

                                        # Now traverse the other synset's hypernyms looking for one of our
                                        # own hypernyms.
                                        otherSyn.traverse( :hypernyms ) {|syn,depth|
                                                if hyperSyns.include?( syn )
                                                        commonSyn = syn
                                                        true
                                                end
                                        }

                                        return commonSyn
                                end
