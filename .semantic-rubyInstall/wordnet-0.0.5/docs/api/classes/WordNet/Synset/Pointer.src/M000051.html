                        def self::parse( pointerString )
                                type, offsetPos, ptrNums = pointerString.split(/\s+/)
                                offset, pos = offsetPos.split( /%/, 2 )
                                new( type, offset, pos, ptrNums[0,2], ptrNums[2,2] )
                        end
