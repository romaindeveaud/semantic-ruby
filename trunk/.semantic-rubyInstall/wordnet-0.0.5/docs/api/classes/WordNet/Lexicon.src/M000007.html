        def familiarity( word, part_of_speech, polyCount=nil )
                wordkey = self.make_word_key( word, part_of_speech )
                return nil unless @index_db.key?( wordkey )
                @index_db[ wordkey ].split( WordNet::SUB_DELIM_RE ).length
        end
