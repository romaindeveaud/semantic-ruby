                def to_s
                        @mutex.synchronize( Sync::SH ) {
                                wordlist = self.words.join(", ").gsub( /%\d/, '' ).gsub( /_/, ' ' )
                                return "#{wordlist} [#{self.part_of_speech}] -- (#{self.gloss})"
                        }
                end
