                def serialize
                        @mutex.synchronize( Sync::SH ) {
                                return [
                                        @filenum,
                                        @wordlist,
                                        @pointerlist,
                                        @frameslist,
                                        @gloss
                                        ].join( WordNet::DELIM )
                                }
                        end
